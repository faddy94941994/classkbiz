            function validate() {
                var validator = new Validator('th');
                validator.setErrorPanel($('#errorText')[0]);
                validator.setSuccessPanel($('#errorText')[0]);
                isNull($('#userName').val(), $('#txtpassword').val());
                validator.addField($('#userName')[0], 'รหัสผู้ใช้งาน', {required: true});
                validator.addField($('#txtpassword')[0], 'รหัสผ่าน', {required: true});
                validator.validateField();
                if (!validator.getDelegator().valid) {
                    validator.displayError();
                    $('#errorText').show();
                    $('#loginTimeOut').hide();
                    GA_login_fail('Please enter Password');
                    return false;
                }

                $.post("analytic.jsp", {cmd: 'c', paObjects: encode64(encryptedElement())});
                $('#errorText').hide();

                return true;
            }

            function validateEmail(value) {
                var emailRegEx = /^[A-Z0-9_-]+(?:\.[A-Z0-9_-]+)*@(?:[A-Z0-9](?:[A-Z0-9-]*[A-Z0-9])?\.)+[A-Z0-9](?:[A-Z0-9-]*[A-Z0-9])$/i;
                return !isValidEmail(value, emailRegEx);
            }

            function isValidEmail(value, regexp) {
                if (value != "" && value != null) {
                    var myRe = regexp.exec(value);
                    if (!myRe) {
                        return false;
                    } else {
                        return true;
                    }
                }
                return true;
            }

                timer = setInterval(function () {

                        'event_category': 'user_login',
                        'event_action': errorEventAction,
                        'event_label': errorMsg
                    });
                }
